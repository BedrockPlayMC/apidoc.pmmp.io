181 public function writeStateToWorld() : void{
182 $world = $this->position->getWorld();
183 $world->getOrLoadChunkAtPosition($this->position)->setFullBlock($this->position->x & Chunk::COORD_MASK, $this->position->y, $this->position->z & Chunk::COORD_MASK, $this->getFullId());
184
185 $tileType = $this->idInfo->getTileClass();
186 $oldTile = $world->getTile($this->position);
187 if($oldTile !== null){
188 if($tileType === null || !($oldTile instanceof $tileType)){
189 $oldTile->close();
190 $oldTile = null;
191 }elseif($oldTile instanceof Spawnable){
192 $oldTile->clearSpawnCompoundCache(); //destroy old network cache
193 }
194 }
195 if($oldTile === null && $tileType !== null){
200 $tile = new $tileType($world, $this->position->asVector3());
201 $world->addTile($tile);
202 }
203 }

722 public function calculateIntercept(Vector3 $pos1, Vector3 $pos2) : ?RayTraceResult{
723 $bbs = $this->getCollisionBoxes();
724 if(count($bbs) === 0){
725 return null;
726 }
727
729 $currentHit = null;
731 $currentDistance = PHP_INT_MAX;
732
733 foreach($bbs as $bb){
734 $nextHit = $bb->calculateIntercept($pos1, $pos2);
735 if($nextHit === null){
736 continue;
737 }
738
739 $nextDistance = $nextHit->hitVector->distanceSquared($pos1);
740 if($nextDistance < $currentDistance){
741 $currentHit = $nextHit;
742 $currentDistance = $nextDistance;
743 }
744 }
745
746 return $currentHit;
747 }
