237 public static function getScheduledTaskTimings(TaskHandler $task, int $period) : TimingsHandler{
238 $name = "Task: " . $task->getTaskName();
239
240 if($period > 0){
241 $name .= "(interval:" . $period . ")";
242 }else{
243 $name .= "(Single)";
244 }
245
246 if(!isset(self::$pluginTaskTimingMap[$name])){
247 self::$pluginTaskTimingMap[$name] = new TimingsHandler($name, self::$schedulerSync, $task->getOwnerName());
248 }
249
250 return self::$pluginTaskTimingMap[$name];
251 }

257 private static function shortenCoreClassName(string $class, string $prefix) : string{
258 if(str_starts_with($class, $prefix)){
259 return (new \ReflectionClass($class))->getShortName();
260 }
261 return $class;

264 public static function getEntityTimings(Entity $entity) : TimingsHandler{
265 if(!isset(self::$entityTypeTimingMap[$entity::class])){
266 if($entity instanceof Player){
267 //the timings viewer calculates average player count by looking at this timer, so we need to ensure it has
268 //a name it can identify. However, we also want to make it obvious if this is a custom Player class.
269 $displayName = $entity::class !== Player::class ? "Player (" . $entity::class . ")" : "Player";
270 }else{
271 $displayName = self::shortenCoreClassName($entity::class, "pocketmine\\entity\\");
272 }
273 self::$entityTypeTimingMap[$entity::class] = new TimingsHandler("Entity Tick - " . $displayName, self::$tickEntity, group: self::GROUP_BREAKDOWN);
274 }
275
276 return self::$entityTypeTimingMap[$entity::class];
277 }
