103 public function coreItemStackToRecipeIngredient(Item $itemStack) : RecipeIngredient{
104 if($itemStack->isNull()){
105 return new RecipeIngredient(null, 0);
106 }
107 if($itemStack->hasAnyDamageValue()){
108 [$id, ] = ItemTranslator::getInstance()->toNetworkId($itemStack->getId(), 0);
109 $meta = 0x7fff;
110 }else{
111 [$id, $meta] = ItemTranslator::getInstance()->toNetworkId($itemStack->getId(), $itemStack->getMeta());
112 }
113 return new RecipeIngredient(new IntIdMetaItemDescriptor($id, $meta), $itemStack->getCount());

116 public function recipeIngredientToCoreItemStack(RecipeIngredient $ingredient) : Item{
117 $descriptor = $ingredient->getDescriptor();
118 if($descriptor === null){
119 return VanillaItems::AIR();
120 }
121 if($descriptor instanceof IntIdMetaItemDescriptor){
122 [$id, $meta] = ItemTranslator::getInstance()->fromNetworkIdWithWildcardHandling($descriptor->getId(), $descriptor->getMeta());
123 return ItemFactory::getInstance()->get($id, $meta, $ingredient->getCount());
124 }
125 if($descriptor instanceof StringIdMetaItemDescriptor){
126 $intId = GlobalItemTypeDictionary::getInstance()->getDictionary()->fromStringId($descriptor->getId());
127 [$id, $meta] = ItemTranslator::getInstance()->fromNetworkIdWithWildcardHandling($intId, $descriptor->getMeta());
128 return ItemFactory::getInstance()->get($id, $meta, $ingredient->getCount());
129 }
130
131 throw new \LogicException("Unsupported conversion of recipe ingredient to core item stack");
132 }

134 public function coreItemStackToNet(Item $itemStack) : ItemStack{
135 if($itemStack->isNull()){
136 return ItemStack::null();
137 }
138 $nbt = null;
139 if($itemStack->hasNamedTag()){
140 $nbt = clone $itemStack->getNamedTag();
141 }
142
143 $isBlockItem = $itemStack->getId() < 256;
144
145 $idMeta = ItemTranslator::getInstance()->toNetworkIdQuiet($itemStack->getId(), $itemStack->getMeta());
146 if($idMeta === null){
147 //Display unmapped items as INFO_UPDATE, but stick something in their NBT to make sure they don't stack with
148 //other unmapped items.
149 [$id, $meta] = ItemTranslator::getInstance()->toNetworkId(ItemIds::INFO_UPDATE, 0);
150 if($nbt === null){
151 $nbt = new CompoundTag();
152 }
153 $nbt->setInt(self::PM_ID_TAG, $itemStack->getId());
154 $nbt->setInt(self::PM_META_TAG, $itemStack->getMeta());
155 }else{
156 [$id, $meta] = $idMeta;
157
158 if($itemStack instanceof Durable && $itemStack->getDamage() > 0){
159 if($nbt !== null){
160 if(($existing = $nbt->getTag(self::DAMAGE_TAG)) !== null){
161 $nbt->removeTag(self::DAMAGE_TAG);
162 $nbt->setTag(self::DAMAGE_TAG_CONFLICT_RESOLUTION, $existing);
163 }
164 }else{
165 $nbt = new CompoundTag();
166 }
167 $nbt->setInt(self::DAMAGE_TAG, $itemStack->getDamage());
168 $meta = 0;
169 }elseif($isBlockItem && $itemStack->getMeta() !== 0){
170 //TODO HACK: This foul-smelling code ensures that we can correctly deserialize an item when the
171 //client sends it back to us, because as of 1.16.220, blockitems quietly discard their metadata
172 //client-side. Aside from being very annoying, this also breaks various server-side behaviours.
173 if($nbt === null){
174 $nbt = new CompoundTag();
175 }
176 $nbt->setInt(self::PM_META_TAG, $itemStack->getMeta());
177 $meta = 0;
178 }
179 }
180
181 $blockRuntimeId = 0;
182 if($isBlockItem){
183 $block = $itemStack->getBlock();
184 if($block->getId() !== BlockLegacyIds::AIR){
185 $blockRuntimeId = RuntimeBlockMapping::getInstance()->toRuntimeId($block->getFullId());
186 }
187 }
188
189 return new ItemStack(
190 $id,
191 $meta,
192 $itemStack->getCount(),
193 $blockRuntimeId,
194 $nbt,
195 [],
196 [],
197 $id === $this->shieldRuntimeId ? 0 : null
198 );
199 }

204 public function netItemStackToCore(ItemStack $itemStack) : Item{
205 if($itemStack->getId() === 0){
206 return VanillaItems::AIR();
207 }
208 $compound = $itemStack->getNbt();
209
210 [$id, $meta] = ItemTranslator::getInstance()->fromNetworkId($itemStack->getId(), $itemStack->getMeta());
211
212 if($compound !== null){
213 $compound = clone $compound;
214 if(($idTag = $compound->getTag(self::PM_ID_TAG)) instanceof IntTag){
215 $id = $idTag->getValue();
216 $compound->removeTag(self::PM_ID_TAG);
217 }
218 if(($damageTag = $compound->getTag(self::DAMAGE_TAG)) instanceof IntTag){
219 $meta = $damageTag->getValue();
220 $compound->removeTag(self::DAMAGE_TAG);
221 if(($conflicted = $compound->getTag(self::DAMAGE_TAG_CONFLICT_RESOLUTION)) !== null){
222 $compound->removeTag(self::DAMAGE_TAG_CONFLICT_RESOLUTION);
223 $compound->setTag(self::DAMAGE_TAG, $conflicted);
224 }
225 }elseif(($metaTag = $compound->getTag(self::PM_META_TAG)) instanceof IntTag){
226 //TODO HACK: This foul-smelling code ensures that we can correctly deserialize an item when the
227 //client sends it back to us, because as of 1.16.220, blockitems quietly discard their metadata
228 //client-side. Aside from being very annoying, this also breaks various server-side behaviours.
229 $meta = $metaTag->getValue();
230 $compound->removeTag(self::PM_META_TAG);
231 }
232 if($compound->count() === 0){
233 $compound = null;
234 }
235 }
236 if($id < -0x8000 || $id >= 0x7fff){
237 throw new TypeConversionException("Item ID must be in range " . -0x8000 . " ... " . 0x7fff . " (received $id)");
238 }
239 if($meta < 0 || $meta >= 0x7fff){ //this meta value may have been restored from the NBT
240 throw new TypeConversionException("Item meta must be in range 0 ... " . 0x7fff . " (received $meta)");
241 }
242
243 try{
244 return ItemFactory::getInstance()->get(
245 $id,
246 $meta,
247 $itemStack->getCount(),
248 $compound
249 );
250 }catch(NbtException $e){
251 throw TypeConversionException::wrap($e, "Bad itemstack NBT data");
252 }
253 }
